33/// A builder that will crate a [`TestContext`].  This builder uses a type-state pattern to ensure
34/// that the require fields are all present.
35///
36/// The fields `timeout`, `rules`, `max_memory`, `max_file_size`, and `max_threads` can be set for
37/// runtime, compile-time, or both.  These methods have a pattern of `run_<field>` for runtime
38/// only, `compile_<field>` for compile-time only, or `<field>` for both runtime and compile-time.
39///
40/// # Usage
41///
42/// ```
43/// # use erudite::{TestContext, FileContent, Rules, MemorySize};
44/// # use std::time::Duration;
45/// # let rules = Rules::new();
46/// let context = TestContext::builder()
47///     .compile_command(["rustc", "-o", "main", "main.rs"])
48///     .run_command(["./main"])   // Run command is always required
49///     .test("group", "hello", "olleh", 1) // At least one test is required
50///     .test("group", "world", "dlrow", 2)
51///     .test("group", "rust", "tsur", 3)
52///     .test("group", "tacocat", "tacocat", 4)
53///     .trim_output(true)
54///     .file(FileContent::string("// some rust code"), "main.rs")
55///     .timeout(Duration::from_secs(5))        // or `run_timeout`/`compile_timeout`
56///     .rules(rules)                           // or `run_rules`/`compile_rules`
57///     .max_memory(MemorySize::from_gib(1))    // or `run_max_memory`/`compile_max_memory`
58///     .max_file_size(MemorySize::from_gib(1)) // or `run_max_file_size`/`compile_max_file_size`
59///     .max_threads(2)                         // or `run_max_threads`/`compile_max_threads`
60///     .build();
61/// ```
62#[derive(Debug, Clone)]
63#[must_use]
64pub struct TestContextBuilder<G, T, Tests = MissingTests, RunCmd = MissingRunCmd> {
65    command: StageConfig<Vec<String>>, // Compile: optional, Run: required
66    test_cases: HashMap<G, Vec<TestCase<T>>>, // required (at least once)
67    trim_output: bool,                 // optional
68    files: Vec<FileConfig>,            // optional
69    timeout: StageConfig<Duration>,    // optional
70    rules: StageConfig<Rules>,         // optional
71    max_memory: StageConfig<MemorySize>, // optional
72    max_file_size: StageConfig<MemorySize>, // optional
73    max_threads: StageConfig<u64>,     // optional
74
75    state: PhantomData<(Tests, RunCmd)>,
76}

116macro_rules! command_config_fns {
117    ($field: ident, $type: ty, $noun: literal) => {
118        concat_idents::concat_idents!(ident = run_, $field {
119            #[doc = concat!("Set the ", $noun, " when running the test")]
120            pub fn ident(mut self, $field: $type) -> Self {
121                self.$field.with_run($field);
122                self
123            }
124        });
125        concat_idents::concat_idents!(ident = compile_, $field {
126            #[doc = concat!("Set the ", $noun, " when compiling the test")]
127            pub fn ident(mut self, $field: $type) -> Self {
128                self.$field.with_compile($field);
129                self
130            }
131        });
132
133        #[doc = concat!("Set the ", $noun, " of _both_ running and compiling the test")]
134        pub fn $field(mut self, $field: $type) -> Self {
135            self.$field.with_both($field);
136            self
137        }
138    };
139}

142impl<G, T, Tests, RunCmd> TestContextBuilder<G, T, Tests, RunCmd> {
143    /// Set whether the tests should trim the output of the program before testing (using
144    /// [`str::trim`])
145    ///
146    /// Default = `false`
147    ///
148    /// ```
149    /// # use erudite::TestContext;
150    /// let ctx = TestContext::builder()
151    ///     .compile_command(["gcc", "-o", "solution", "solution.c"])
152    ///     .run_command(["solution.c"])
153    ///     .test("group", "hello world", "dlrow olleh", true)
154    ///     .trim_output(true)
155    ///     .build();
156    /// ```
157    pub fn trim_output(mut self, trim_output: bool) -> Self {
158        self.trim_output = trim_output;
159        self
160    }
161
162    /// Set the command used for compilation of the test.  If this function is not called, then
163    /// _no_ compilation step will be performed, even if the other compile settings are used.
164    ///
165    /// The command must have at least one item to be valid.  If it does not, the error will be
166    /// propogated when calling [`TestRunner::compile`]
167    ///
168    /// [`TestRunner::compile`]: crate::runner::TestRunner::compile
169    ///
170    /// ```
171    /// # use erudite::{FileContent, TestContext};
172    /// let ctx = TestContext::builder()
173    ///     .compile_command(["gcc", "-o", "solution", "solution.c"])
174    ///     .run_command(["solution.c"])
175    ///     .test("group", "hello world", "dlrow olleh", true)
176    ///     .build();
177    /// ```
178    pub fn compile_command(
179        mut self,
180        compile_command: impl IntoIterator<Item = impl Into<String>>,
181    ) -> Self {
182        self.command
183            .with_compile(compile_command.into_iter().map(Into::into).collect());
184        self
185    }
186
187    /// Add a file to be inserted into the runtime environment of the test.  This is added before
188    /// compilation, so it works well for libraries or input/output manipulation.
189    ///
190    /// If `source` is a path, the file will only be copied when a test is compiled, meaning
191    /// that if the file changes or is removed, then the output may differ between two
192    /// instances of the test runner from a single context.
193    ///
194    /// If the intended behaviour is to read the file _now_, consider reading the file directly
195    /// and placing it into [`FileContent::bytes`].
196    ///
197    /// `destination` is path relative to the directory used for the test environment.  If
198    /// destination an absolute path, then it will be made relative to the test environment, i.e.,
199    /// `/foo/bar` -> `<test-env>/foo/bar`.
200    ///
201    /// ```
202    /// # use erudite::{FileContent, TestContext};
203    /// let ctx = TestContext::builder()
204    ///     .run_command(["node", "solution.js"])
205    ///     .test("group", "hello world", "dlrow olleh", true)
206    ///     .file(FileContent::string("// some javascript code"), "solution.js")
207    ///     .file(FileContent::path("/foo/bar"), "some_other_file")
208    ///     .build();
209    /// ```
210    pub fn file(mut self, source: impl Into<FileContent>, destination: impl AsRef<Path>) -> Self {
211        self.files.push(FileConfig::new(source, destination));
212        self
213    }
214
215    /// Add multiple files to be inserted into the runtime environment of the test.  These files
216    /// are added before compilation, so it works well for libraries or input/output manipulation.
217    ///
218    /// If `source` is a path, the file will only be copied when a test is compiled, meaning
219    /// that if the file changes or is removed, then the output may differ between two
220    /// instances of the test runner from a single context.
221    ///
222    /// If the intended behaviour is to read the file _now_, consider reading the file directly
223    /// and placing it into [`FileContent::bytes`].
224    ///
225    /// `destination` is path relative to the directory used for the test environment.  If
226    /// destination an absolute path, then it will be made relative to the test environment, i.e.,
227    /// `/foo/bar` -> `<test-env>/foo/bar`.
228    ///
229    /// ```
230    /// # use erudite::{FileContent, FileConfig, TestContext};
231    /// let ctx = TestContext::builder()
232    ///     .run_command(["node", "solution.js"])
233    ///     .test("group", "hello world", "dlrow olleh", true)
234    ///     .files([
235    ///         FileConfig::new(FileContent::string("// some javascript code"), "solution.js"),
236    ///         FileConfig::new(FileContent::path("/foo/bar"), "some_other_file")
237    ///     ])
238    ///     .build();
239    /// ```
240    pub fn files(mut self, files: impl IntoIterator<Item = impl Into<FileConfig>>) -> Self {
241        self.files.extend(files.into_iter().map(Into::into));
242        self
243    }
244
245    command_config_fns!(rules, Rules, "rules for execution");
246    command_config_fns!(max_memory, MemorySize, "maximum memory usage");
247    command_config_fns!(max_file_size, MemorySize, "maximum file size");
248    command_config_fns!(max_threads, u64, "max thread count");
249    command_config_fns!(timeout, Duration, "timeout");
250}

278impl<G, T, Tests, RunCmd> TestContextBuilder<G, T, Tests, RunCmd>
279where
280    T: Clone, // This bound is a little bit arbitrary here, but it makes errors cleaner as
281    // [`TestContext::test_builder`] needs it.
282    G: Hash + Eq,
283{
284    /// Add a single test to this context in a group
285    ///
286    /// The `data` argument is for any data that you wish to associate with this test (on top of
287    /// its index, which is assocated by default).  This is useful for adding additional
288    /// information such as visiblity to the test.
289    ///
290    /// The data is accessible by [`TestRunner::filter_tests`] and [`TestResult::data`].
291    ///
292    /// The `group` argument is used to associate tests with eachother.  When creating a test
293    /// runner, one uses the `group` to select just the tests that they wish to run.  Group keys
294    /// must implement [`Hash`] and [`Eq`], but other than that, there is no restriction.  This
295    /// function requires an owned `G` for each test case, but to reduce copies,
296    /// [`TestContextBuilder::tests`] may be preferred.
297    ///
298    /// There is a special case if `G` is `()`: a [`TestContext::default_test_runner`] becomes
299    /// available, which requires no argment and does not return an option.  This is ideal if only
300    /// one test group is necessary.
301    ///
302    /// [`TestRunner::filter_tests`]: crate::runner::TestRunner::filter_tests
303    /// [`TestResult::data`]: crate::runner::TestResult::data
304    ///
305    /// ```
306    /// # use erudite::TestContext;
307    /// let ctx = TestContext::builder()
308    ///     .run_command(["echo", "hi"])
309    ///     .test("group", "hello world", "dlrow olleh", true)
310    ///     .build();
311    /// ```
312    pub fn test(
313        mut self,
314        group: G,
315        input: impl Into<String>,
316        output: impl Into<ExpectedOutput>,
317        data: T,
318    ) -> TestContextBuilder<G, T, SetTests, RunCmd> {
319        self.test_cases
320            .entry(group)
321            .or_default()
322            .push(TestCase::new(input, output, data));
323        self.transform()
324    }
325
326    /// Add a series of tests to this context
327    ///
328    /// The `group` argument is used to associate tests with eachother.  When creating a test
329    /// runner, one uses the `group` to select just the tests that they wish to run.  Group keys
330    /// must implement [`Hash`] and [`Eq`], but other than that, there is no restriction.  This
331    /// function may be preferable over [`TestContextBuilder::test`] as there is only need for one
332    /// allocated group key.
333    ///
334    /// There is a special case if `G` is `()`: a [`TestContext::default_test_runner`] becomes
335    /// available, which requires no argment and does not return an option.  This is ideal if only
336    /// one test group is necessary.
337    ///
338    /// See [`Self::test`] for additional information.
339    /// ```
340    /// # use erudite::TestContext;
341    /// let ctx = TestContext::builder()
342    ///     .run_command(["echo", "hi"])
343    ///     .tests("group", [
344    ///         ("hello world", "dlrow olleh", true),
345    ///         ("good morning", "gninrom doog", true),
346    ///     ])
347    ///     .build();
348    /// ```
349    // NOTE: this function in the type-state builder is not perfect, as the caller could pass an
350    // iterator with 0 elements, but it's a relatively rare case.
351    pub fn tests(
352        mut self,
353        group: G,
354        tests: impl IntoIterator<Item = impl Into<TestCase<T>>>,
355    ) -> TestContextBuilder<G, T, SetTests, RunCmd> {
356        self.test_cases
357            .entry(group)
358            .or_default()
359            .extend(tests.into_iter().map(Into::into));
360        self.transform()
361    }
362
363    /// Add several test groups to this context
364    ///
365    /// The `group` argument is used to associate tests with eachother.  When creating a test
366    /// runner, one uses the `group` to select just the tests that they wish to run.  Group keys
367    /// must implement [`Hash`] and [`Eq`], but other than that, there is no restriction.  This
368    /// function may be preferable over [`TestContextBuilder::test`] as there is only need for one
369    /// allocated group key.
370    ///
371    /// There is a special case if `G` is `()`: a [`TestContext::default_test_runner`] becomes
372    /// available, which requires no argment and does not return an option.  This is ideal if only
373    /// one test group is necessary.
374    ///
375    /// NOTE: This gets very syntactically noisy when writing out by hand.  This function is very
376    /// much intended for use with iterators, rather than manually written.  If you're manully
377    /// writing cases, see [`TestContextBuilder::tests`].
378    ///
379    /// ```
380    /// # use erudite::TestContext;
381    /// let ctx = TestContext::builder()
382    ///     .run_command(["echo", "hi"])
383    ///     .test_groups([
384    ///         ("group1", [("hello world", "dlrow olleh", true), ("good morning", "gninrom doog", true)]),
385    ///         ("group2", [("52", "even", true), ("27", "odd", true)])
386    ///     ])
387    ///     .build();
388    /// ```
389    // NOTE: this function in the type-state builder is not perfect, as the caller could pass an
390    // iterator with 0 elements, but it's a relatively rare case.
391    pub fn test_groups(
392        mut self,
393        test_groups: impl IntoIterator<Item = (G, impl IntoIterator<Item = impl Into<TestCase<T>>>)>,
394    ) -> TestContextBuilder<G, T, SetTests, RunCmd> {
395        test_groups.into_iter().for_each(|(group, tests)| {
396            self.test_cases
397                .entry(group)
398                .or_default()
399                .extend(tests.into_iter().map(Into::into));
400        });
401        self.transform()
402    }
403}

405impl<G, T> TestContextBuilder<G, T, SetTests, SetRunCmd>
406where
407    G: Hash + Eq,
408{
409    /// Finish building this test context
410    pub fn build(self) -> TestContext<G, T> {
411        // Just some sanity checks
412        assert!(self.command.run().is_some());
413        assert!(!self.test_cases.is_empty());
414        TestContext {
415            trim_output: self.trim_output,
416            files: self.files,
417            test_cases: self
418                .test_cases
419                .into_iter()
420                .map(|(k, v)| (k, v.into_boxed_slice().into()))
421                .collect(),
422            command: self.command.into(),
423            timeout: self.timeout,
424            rules: self.rules,
425            max_memory: self.max_memory,
426            max_file_size: self.max_file_size,
427            max_threads: self.max_threads,
428        }
429    }
430}
