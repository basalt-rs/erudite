6/// Output that is expected from a test
7///
8/// If the `regex` feature is enabled, this can also match regular expressions
9#[derive(Debug, Clone, From)]
10pub enum ExpectedOutput {
11    /// Match a case-sensitive string
12    #[from]
13    String(String),
14    /// Match a case-insensitive string
15    StringInsensitive(String),
16    /// Match a regular expression if the `regex` feature is enabled
17    #[cfg(feature = "regex")]
18    #[from]
19    Regex(regex::Regex),
20}
21
22impl From<&str> for ExpectedOutput {
23    fn from(value: &str) -> Self {
24        ExpectedOutput::String(value.into())
25    }
26}
27
28impl PartialEq for ExpectedOutput {
29    fn eq(&self, other: &Self) -> bool {
30        match (self, other) {
31            (ExpectedOutput::String(s), ExpectedOutput::String(o)) => s == o,
32            (ExpectedOutput::StringInsensitive(s), ExpectedOutput::StringInsensitive(o)) => s == o,
33            // NOTE: this is a little weird, as two different regex strings can match the same
34            // patterns, i.e., `..*` and `.+`, but it's good enough
35            #[cfg(feature = "regex")]
36            (ExpectedOutput::Regex(s), ExpectedOutput::Regex(o)) => s.as_str() == o.as_str(),
37            _ => false,
38        }
39    }
40}
41
42impl ExpectedOutput {
43    /// Check whether a given string matches this expected output
44    pub fn is_valid(&self, output: &str) -> bool {
45        match self {
46            Self::String(ref s) => s == output,
47            Self::StringInsensitive(ref s) => s.eq_ignore_ascii_case(output),
48            #[cfg(feature = "regex")]
49            Self::Regex(ref reg) => reg.is_match(output),
50        }
51    }
52}

54/// A test case which contains input, output, and some associated data
55#[derive(Debug, Clone, PartialEq)]
56pub struct TestCase<T = ()> {
57    input: String,
58    output: ExpectedOutput,
59    data: T,
60}
61
62impl<T> TestCase<T> {
63    /// Construct a new test case
64    ///
65    /// If using [`TestContextBuilder`], consider using the from implementation for `(I, O)` or
66    /// `(I, O, T)` tuples.
67    ///
68    /// [`TestContextBuilder`]: crate::context::TestContextBuilder
69    pub fn new(input: impl Into<String>, output: impl Into<ExpectedOutput>, data: T) -> Self {
70        Self {
71            input: input.into(),
72            output: output.into(),
73            data,
74        }
75    }
76
77    /// Retrieve the input value associated with this test case
78    pub fn input(&self) -> &str {
79        &self.input
80    }
81
82    /// Retrieve the expected output for this test case
83    pub fn output(&self) -> &ExpectedOutput {
84        &self.output
85    }
86
87    /// Get the data assocated with this test case
88    ///
89    /// See also: [`TestCase::into_data`]
90    pub fn data(&self) -> &T {
91        &self.data
92    }
93
94    /// Get the owned data of this test case, consuming the case
95    ///
96    /// See also: [`TestCase::data`]
97    pub fn into_data(self) -> T {
98        self.data
99    }
100}

102impl<I, O> From<(I, O)> for TestCase<()>
103where
104    I: Into<String>,
105    O: Into<ExpectedOutput>,
106{
107    fn from((input, output): (I, O)) -> Self {
108        Self::new(input, output, ())
109    }
110}
111
112impl<I, O, T> From<(I, O, T)> for TestCase<T>
113where
114    I: Into<String>,
115    O: Into<ExpectedOutput>,
116{
117    fn from((input, output, data): (I, O, T)) -> Self {
118        Self::new(input, output, data)
119    }
120}

134    #[test]
135    fn expected_output_equality() {
136        assert_eq!(
137            ExpectedOutput::from("foobar"),
138            ExpectedOutput::from("foobar"),
139        );
140        assert_ne!(
141            ExpectedOutput::from("foobar"),
142            ExpectedOutput::from("bazqux"),
143        );
144        assert_ne!(
145            ExpectedOutput::from("foobar"),
146            ExpectedOutput::from(re!(".")),
147        );
148        assert_ne!(
149            ExpectedOutput::from(re!(".")),
150            ExpectedOutput::from("foobar"),
151        );
152        assert_eq!(
153            ExpectedOutput::from(re!(".")),
154            ExpectedOutput::from(re!(".")),
155        );
156        assert_ne!(
157            ExpectedOutput::from(re!(".")),
158            ExpectedOutput::from(re!(".+")),
159        );
160    }

162    #[test]
163    fn expect_output_is_valid_string() {
164        let validator = ExpectedOutput::from("hi");
165        assert!(validator.is_valid("hi"));
166        assert!(!validator.is_valid("bye"));
167    }
168
169    #[test]
170    fn expect_output_is_valid_empty_string() {
171        let validator = ExpectedOutput::String(String::new());
172        assert!(validator.is_valid(""));
173        assert!(!validator.is_valid("this is not empty"));
174    }
175
176    #[test]
177    fn expect_output_is_valid_regex() {
178        let validator = ExpectedOutput::from(re!(r"([01]\d|2[0-3])(:[0-5]\d){2}"));
179        assert!(validator.is_valid("04:22:57"));
180        assert!(validator.is_valid("14:22:57"));
181        assert!(!validator.is_valid("24:22:57"));
182    }

184    #[test]
185    fn test_case_getters() {
186        let case = TestCase::new("hello", "world", 69);
187        assert_eq!(case.input(), "hello");
188        assert!(matches!(case.output(), ExpectedOutput::String(x) if x == "world"));
189        assert_eq!(*case.data(), 69);
190        assert_eq!(case.into_data(), 69);
191        let case2 = TestCase::new("foo", "bar", 420);
192        assert_eq!(case2.input(), "foo");
193        assert!(matches!(case2.output(), ExpectedOutput::String(x) if x == "bar"));
194        assert_eq!(*case2.data(), 420);
